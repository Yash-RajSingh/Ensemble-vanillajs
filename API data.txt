Register API:-
    Base URL:- https://ensemble-p2.herokuapp.com/authentication/register
    arguements:- username, email,password,sec_question,sec_answer.
    response:- message, status.

Login API:-
    Base URL:- https://ensemble-p2.herokuapp.com/authentication/login
    arguements:- email,password.
    response:- message, status, token, username, unique user ID.
